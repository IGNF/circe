93 circe --sourceCRS=NTFLAMB3. --sourceFormat=IEN.METERS. --targetCRS=WGS84G. --targetFormat=ILP.METERS.DEGREES --metadataFile=data/IGNF.xml --sourcePathname=./datafile.txt --targetPathname=./outfile.txt\n\
